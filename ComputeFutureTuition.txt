
Start

Set starting tuition to $10,000

Iterate loop through ten years:
	Increase tuition by 5% each year
	Initialize total cost to 0
	Iterate through four years after the tenth year:
	Add tuition to total cost
	Increase tuition by 5%

Display the result:
	Display tuition cost in ten years with 2 decimal places
	Display total cost of four years' worth of tuition after the tenth year with 2 decimal places

End


----------------------------


import java.util.*;

  public class ComputeFutureTuition {
    public static void main(String[] args) {
    // Sets starting tuition
    double tuition = 10500;

    // Calculates tuition in ten years
    for (int i = 1; i <= 10; i++) {
    // Increases tuition rate by five percent each year
      tuition *= 1.05;
    }

  // Calculates total cost of four years' tuition after ten years
    double totalCost = 0;
    for (int i = 0; i < 4; i++) {
      totalCost += tuition;
      tuition *= 1.05;
    }

// Displays the result to the user
      System.out.printf("Tuition cost in 10 years: $%.2f\n", tuition);
          System.out.printf("\nTotal cost of 4 years' worth of tuition after the 10th year: $%.2f\n", totalCost);
      
     }
    
  }
